\section{Période I}
\paragraph{I-1 Traité de Tordesillas} (\emph{hist. $1494$}) (ESP,
POR)\\
\noindent Partage du monde entre le Portugal (Brésil, Afrique, Asie)
et l'Espagne (le reste).

\paragraph{I-2 Élection au Saint-Empire} (\emph{hist. $1519$) (ESP,
  FRA, pol, ang, rus)\\
\noindent Mise pour devenir empereur, recommence à la mort de
l'empereur (si non espagnol).

\paragraph{I-3 Conflit commercial vénéto-turc} (\emph{sans date
  précise}) (VEN, TUR)\\
\noindent CB commercial ou gratuit de Venise sur la Turquie autour du
centre de commerce grand Orient.

\paragraph{I-4 Guerres d'Italie} (\emph{hist. $1494-1505$,
  $1510-1514$, $1515-1516$}) (FRA, ESP, ven) ($2$)\\
\noindent 1/ CB de FRA contre Naples (et son propriétaire).\\
\noindent 2/ CB de FRA contre Milan (et son propriétaire).

\paragraph{I-5 Déclaration de la liberté hongroise/Mariage de Bohème}
(\emph{hist. $1505$/$1526$}) (HAB) ($2$)\\
\noindent 1/ CB de HAB contre Hongrie, blocage de l'héritage
hongrois.\\
\noindent 2/ La Bohème est annexée au HAB.

\paragraph{I-6 Mariage hongrois} (\emph{hist. $1491$}) (HAB)\\
\noindent Annexion de la Hongrie par HAB possible en période II.

\paragraph{I-7 Mariage milanais} (\emph{hist. $?$}) (HAB)\\
\noindent Milan devient vassal spécial de HAB qui pourra le céder à
ESP (action dynastique).

\paragraph{I-8 Réforme} (\emph{hist. $1522-1560$}) (ESP, FRA, ANG,
POL, rus, sue) ($3$)\\
\noindent 1/ Certains mineurs deviennent protestants, POL choisi son
attitude face à l'othodoxie, RUS choisi son attitude religieuse.\\
\noindent 2/ ESP, FRA, ANG et POL choississent leur attitude face à la
Réforme (protestant, conciliabnt ou contre-réforme).\\
\noindent 3/ Certains mineurs deviennent protestants.

\paragraph{I-9 Pressions diplomatiques turques} (\emph{Sans date
  précise}) (TUR) ($*$)\\
\noindent Bonus diplomatique pour TUR à ce tour.

\paragraph{I-10 Guerre contre l'Écosse} (\emph{hist. $1513-1514$})
(ANG)\\
\noindent L'Écosse déclare la guerre à ANG.

\paragraph{I-11 Chûte du khanat de la Horde d'or/Annexion russes}
(\emph{hist. $1502$/$1510$ et $1517$}) (RUS)\\
\noindent 1/ Guerre Crimée-Horde d'or, la Horde perd son pion armée.\\
\noindent 2/ RUS peut annexer Pskow ou Ryazan, CB de POL contre RUS.

\paragraph{I-12 Action dynastique Habsbourg} (\emph{différentes
  dates}) (ESP)\\
\noindent Déclenche une action dynastique au choix de ESP

\paragraph{I-A Héritage Habsbourg} (\emph{hist. $1502$}) (ESP)
(déclenché par action dynastique)\\
\noindent HAB devient vassal spécial de ESP et ESP peut annexer les
provicness du Nord-Est.

\paragraph{I-B Héritage bourguignon} (\emph{hist. $1506$}) (ESP)
(déclenché par action dynastique)\\
\noindent ESP annexe le mineur Bourgogne.

\paragraph{I-C Héritage de la Bohème} (\emph{hist. $1526$}) (HAB)
(déclenché par I-5 (2) ou action dynastique)\\
\noindent HAB annexe la Bohème.

\paragraph{I-D Héritage hongrois} (\emph{hist. $?$}) (HAB) (déclenché
en période II si I-6 ou action dynastique actif)\\
\noindent HAB annexe la Hongrie.

\paragraph{I-E Partage de la Hongrie} (\emph{hist. $1526$}) (HAB, TUR,
pol) (déclenché en cas de victoire majeure de TUR (ou HAB) sur
Hongrie)\\
\noindent La Hongrie est partagée entre HAB, TUR, POL et Transylvanie.

\paragraph{I-F Héritage de Milan} (\emph{hist. $?$}) (HAB, ESP)
(déclenché par I-7 ou action dynastique)\\
\noindent Milan devient vassal spécial de HAB qui pourra le céder à
ESP (action dynastique).

\paragraph{I-G Milanais espagnol} (\emph{hist. $?$}) (ESP) (déclenché
par action dynastique)\\
\noindent ESP annexe Milan.

\paragraph{I-H Sécularisation de l'ordre des chevaliers teutoniques}
(\emph{hist. $1525$}) (POL)\\
\noindent Les chevaliers nordiques perdent une armée et des
provinces. Les provinces vont au duché de Prusse qui est annexé par
POL.
